    import "log"

    const debug debugging = true // or flip to false

    type debugging bool

    func (d debugging) Printf(format string, args ...interface{}) {
        d {
            log.Printf(format, args...)
        }
    }

    func main() {
        debug.Printf("foo %d %.2f", 42, 12.7)
