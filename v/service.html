interface Service #

interface Service {
	// Start signals to the service manager the given service should start.

	start() !

	// Stop signals to the service manager the given service should stop.
	stop() !

	// Restart signals to the service manager the given service should stop then start.
	restart() !

	// Down prevents a service from starting at boot
	// down() !
	// Install setups up the given service in the OS service manager. This may require
	// greater rights. Will return an error if it is already installed.
	install() !

	// Uninstall removes the given service from the OS service manager. This may require
	// greater rights. Will return an error if the service is not present.
	uninstall() !

	// Status returns the current service status.
	status() Status

	// Enabled checks if service is normally up
	enabled() bool
}

Service represents a service that can be run or controlled.
